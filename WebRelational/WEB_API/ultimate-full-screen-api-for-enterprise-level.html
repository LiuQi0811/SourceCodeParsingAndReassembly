📝 CSS 代码

:fullscreen 示例 
/* 标准语法 */
.fullscreen-styled:fullscreen {
    display: flex;
    flex-direction: column;
    justify-content: center;
    align-items: center;
    background: linear-gradient(145deg, #667eea, #764ba2);
    color: white;
}

/* WebKit 前缀 */
.fullscreen-styled:-webkit-full-screen {
    display: flex;
    flex-direction: column;
    justify-content: center;
    align-items: center;
    background: linear-gradient(145deg, #667eea, #764ba2);
    color: white;
}

/* Firefox 前缀 */
.fullscreen-styled:-moz-full-screen {
    display: flex;
    flex-direction: column;
    justify-content: center;
    align-items: center;
    background: linear-gradient(145deg, #667eea, #764ba2);
    color: white;
}

/* IE/Edge 前缀 */
.fullscreen-styled:-ms-fullscreen {
    display: flex;
    flex-direction: column;
    justify-content: center;
    align-items: center;
    background: linear-gradient(145deg, #667eea, #764ba2);
    color: white;
}
