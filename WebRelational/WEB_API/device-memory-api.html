// Adaptive loading based on device memory function getAssetQuality() { const memory = navigator.deviceMemory || 4; if (memory < 2) { return 'low'; // Low-end devices } else if (memory < 4) { return 'medium'; // Mid-range devices } else { return 'high'; // High-end devices } } // Adjust worker count based on hardware function getOptimalWorkerCount() { const cores = navigator.hardwareConcurrency || 4; const memory = navigator.deviceMemory || 4; // Use fewer workers on low-memory devices const memoryFactor = Math.min(memory / 4, 1); return Math.max(1, Math.floor(cores * memoryFactor)); } // Monitor memory usage function checkMemoryPressure() { if (performance.memory) { const used = performance.memory.usedJSHeapSize; const limit = performance.memory.jsHeapSizeLimit; const ratio = used / limit; if (ratio > 0.9) { console.warn('Memory pressure high:', ratio.toFixed(2)); // Trigger cleanup } } }

Device Memory API

navigator.deviceMemory Property

Returns the approximate amount of device memory in gigabytes. The value is a float that can be 0.25, 0.5, 1, 2, 4, or 8 (or higher).

Related APIs

navigator.hardwareConcurrency Property

Returns the number of logical processor cores available to run threads on the user's computer.

performance.memory Property

Returns an object with JavaScript heap memory information (Chrome only). Contains usedJSHeapSize, totalJSHeapSize, and jsHeapSizeLimit.

performance.memory.usedJSHeapSize Property

The amount of memory currently being used by the JavaScript heap.

performance.memory.totalJSHeapSize Property

The total allocated heap size.

performance.memory.jsHeapSizeLimit Property

The maximum size of the JavaScript heap that can be allocated.

Memory Values

Possible deviceMemory Values Enum

0.25, 0.5, 1, 2, 4, 8 (GB) - The API rounds down to protect user privacy.

Usage Example

// Check if API is supported if ('deviceMemory' in navigator) { const memory = navigator.deviceMemory; console.log(`Device has approximately ${memory} GB of RAM`); // Adapt application behavior if (memory < 2) { // Low-end device: use simpler UI, fewer workers console.log('Low-end device detected'); } else if (memory >= 8) { // High-end device: can use more resources console.log('High-end device detected'); } } // Combine with hardwareConcurrency const cores = navigator.hardwareConcurrency || 4; const memory = navigator.deviceMemory || 4; // Calculate optimal worker count const workerCount = Math.min( cores - 1, // Leave one core for main thread Math.floor(memory) // Scale with memory ); console.log(`Using ${workerCount} workers`);
