SVG API 参考

                

                

                    

                        
基础图形元素

                        

                            
<rect>

                            
矩形元素，用于绘制矩形和正方形

                            
属性: x, y, width, height, rx, ry

                        

                        

                            
<circle>

                            
圆形元素，用于绘制圆形

                            
属性: cx, cy, r

                        

                        

                            
<ellipse>

                            
椭圆元素，用于绘制椭圆形

                            
属性: cx, cy, rx, ry

                        

                        

                            
<line>

                            
线条元素，用于绘制直线

                            
属性: x1, y1, x2, y2

                        

                        

                            
<polyline>

                            
折线元素，用于绘制连接的线段

                            
属性: points

                        

                        

                            
<polygon>

                            
多边形元素，用于绘制封闭的多边形

                            
属性: points

                        

                    


                    

                        
路径元素

                        

                            
<path>

                            
路径元素，SVG中最强大的绘图元素

                            
属性: d (路径命令)

                        

                        

                            
路径命令

                            
M=移动, L=直线, H=水平线, V=垂直线, C=三次贝塞尔, S=平滑三次贝塞尔, Q=二次贝塞尔, T=平滑二次贝塞尔, A=弧线, Z=闭合路径

                            
示例: M 10 10 L 100 10 L 100 100 Z

                        

                    


                    

                        
变换属性

                        

                            
transform

                            
应用2D或3D变换到元素

                            
transform="translate(tx ty) | rotate(angle) | scale(sx sy) | skewX(angle) | skewY(angle) | matrix(a b c d e f)"

                        

                    


                    

                        
渐变元素

                        

                            
<linearGradient>

                            
线性渐变定义

                            
属性: x1, y1, x2, y2, gradientUnits, spreadMethod

                        

                        

                            
<radialGradient>

                            
径向渐变定义

                            
属性: cx, cy, r, fx, fy, gradientUnits, spreadMethod

                        

                        

                            
<stop>

                            
渐变停止点

                            
属性: offset, stop-color, stop-opacity

                        

                    


                    

                        
滤镜效果

                        

                            
<filter>

                            
滤镜容器元素

                            
属性: id, x, y, width, height, filterUnits

                        

                        

                            
<feGaussianBlur>

                            
高斯模糊滤镜

                            
属性: in, stdDeviation

                        

                        

                            
<feDropShadow>

                            
投影滤镜

                            
属性: dx, dy, stdDeviation, flood-color, flood-opacity

                        

                        

                            
<feColorMatrix>

                            
颜色矩阵滤镜

                            
属性: type, values

                        

                    


                    

                        
图案与遮罩

                        

                            
<pattern>

                            
图案定义元素

                            
属性: id, x, y, width, height, patternUnits, patternTransform

                        

                        

                            
<clipPath>

                            
裁剪路径定义

                            
属性: id, clipPathUnits

                        

                        

                            
<mask>

                            
遮罩定义

                            
属性: id, x, y, width, height, maskUnits, maskContentUnits

                        

                    


                    

                        
文本元素

                        

                            
<text>

                            
文本元素

                            
属性: x, y, dx, dy, text-anchor, font-family, font-size, font-weight, font-style

                        

                        

                            
<tspan>

                            
文本片段，用于文本内的样式变化

                            
属性: x, y, dx, dy

                        

                        

                            
<textPath>

                            
沿路径排列文本

                            
属性: href, startOffset, method, spacing

                        

                    


                    

                        
动画元素

                        

                            
<animate>

                            
属性动画

                            
属性: attributeName, from, to, dur, repeatCount, fill, calcMode, keyTimes, keySplines

                        

                        

                            
<animateTransform>

                            
变换动画

                            
属性: type, from, to, dur, repeatCount

                        

                        

                            
<animateMotion>

                            
沿路径运动动画

                            
属性: path, dur, repeatCount, rotate

                        

                        

                            
<set>

                            
设置属性值动画

                            
属性: attributeName, to, begin, end

                        

                    


                    

                        
图像与嵌入

                        

                            
<image>

                            
嵌入图像

                            
属性: href, x, y, width, height, preserveAspectRatio

                        

                        

                            
<use>

                            
引用复用元素

                            
属性: href, x, y, width, height

                        

                        

                            
<symbol>

                            
可复用图形模板

                            
属性: id, viewBox, preserveAspectRatio

                        

                    


                    

                        
容器元素

                        

                            
<svg>

                            
SVG根元素或嵌套SVG

                            
属性: viewBox, width, height, preserveAspectRatio, xmlns

                        

                        

                            
<g>

                            
分组元素，用于组织图形

                            
属性: id, transform, 以及其他可继承属性

                        

                        

                            
<defs>

                            
定义容器，用于存储可复用元素

                            
无特殊属性

                        

                    


                    

                        
通用属性

                        

                            
fill

                            
填充颜色或渐变

                            
值: 颜色值 | url(#gradientId) | none

                        

                        

                            
stroke

                            
描边颜色

                            
值: 颜色值 | url(#gradientId) | none

                        

                        

                            
stroke-width

                            
描边宽度

                            
值: 长度值 (px, em, %等)

                        

                        

                            
opacity

                            
整体透明度

                            
值: 0.0 - 1.0

                        

                        

                            
fill-opacity / stroke-opacity

                            
填充/描边透明度

                            
值: 0.0 - 1.0

                        

                        

                            
stroke-linecap

                            
线条端点样式

                            
值: butt | round | square

                        

                        

                            
stroke-linejoin

                            
线条连接样式

                            
值: miter | round | bevel

                        

                        

                            
stroke-dasharray

                            
虚线样式

                            
值: 长度序列 (如: 5,10,5)
