什么是 Shared Storage API？

Shared Storage API 是一种客户端存储机制，它允许在跨站点上下文中进行未分区的数据访问，同时保护用户隐私。这是 Privacy Sandbox 计划的一部分，旨在替代第三方 Cookie，提供更隐私友好的广告和个性化解决方案。

核心特性：Shared Storage API 提供了一种安全的方式来存储和访问跨站点数据，而不会泄露用户的具体浏览历史或个人信息。所有数据读取操作都在隔离的 Worklet 环境中执行。

主要优势

  • 隐私保护

    数据读取在隔离的 Worklet 中进行，无法直接访问原始数据，防止跨站点追踪

  • 跨站点共享

    同一来源的数据可以在多个站点间安全共享，支持广告频次控制等用例

  • 灵活的操作类型

    支持 run() 和 selectURL() 两种操作类型，满足不同的业务需求

  • Private Aggregation 集成

    可与 Private Aggregation API 配合使用，进行隐私安全的聚合报告

clear()

Async

什么是 Worklet？

创建 Worklet 模块

完整 Worklet 模块示例
// worklet-module.js - Worklet 模块文件

// 频次控制操作
class FrequencyCapOperation extends SharedStorageRunOperation {
  async run(data) {
    const { adId, maxViews } = data;
    const key = `ad-views-${adId}`;
    
    // 读取当前观看次数
    const currentViews = await this.sharedStorage.get(key);
    const views = currentViews ? parseInt(currentViews) : 0;
    
    // 检查是否超过限制
    if (views >= maxViews) {
      return { 
        canShow: false, 
        reason: 'frequency-cap-exceeded' 
      };
    }
    
    // 增加计数
    await this.sharedStorage.set(key, String(views + 1));
    
    return { 
      canShow: true, 
      currentViews: views + 1,
      maxViews 
    };
  }
}

// 创意选择操作
class CreativeRotationOperation extends SharedStorageSelectURLOperation {
  async run(data) {
    const { campaignId } = data;
    const key = `rotation-${campaignId}`;
    
    // 获取当前索引
    const lastIndex = await this.sharedStorage.get(key);
    const currentIndex = lastIndex ? parseInt(lastIndex) : 0;
    
    // 计算下一个索引（轮询）
    const nextIndex = (currentIndex + 1) % this.urls.length;
    
    // 保存新索引
    await this.sharedStorage.set(key, String(nextIndex));
    
    return currentIndex;
  }
}

// A/B 测试操作
class ABTestOperation extends SharedStorageSelectURLOperation {
  async run(data) {
    const { experimentId } = data;
    const key = `experiment-${experimentId}`;
    
    // 检查是否已分配组
    let groupIndex = await this.sharedStorage.get(key);
    
    if (groupIndex === undefined) {
      // 首次访问，随机分配
      groupIndex = Math.floor(Math.random() * this.urls.length);
      await this.sharedStorage.set(key, String(groupIndex));
    }
    
    return parseInt(groupIndex);
  }
}

// 注册所有操作
register('frequency-cap', FrequencyCapOperation);
register('creative-rotation', CreativeRotationOperation);
register('ab-test', ABTestOperation);

    urls,

      resolveToConfig: true
    }

参数说明

参数 类型 描述

options.resolveToConfig boolean 是否返回 FencedFrameConfig

完整示例

// Worklet 模块
class CheckKnownCustomer extends SharedStorageRunOperation {
  async run(data) {
    const { userId } = data;
    
    // 检查用户是否为已知客户
    const knownCustomer = await this.sharedStorage.get(`user-${userId}`);
    
    if (knownCustomer) {
      const userData = JSON.parse(knownCustomer);
      return {
        isKnown: true,
        tier: userData.tier,
        lastVisit: userData.lastVisit
      };
    }
    
    return { isKnown: false };
  }
}

register('check-known-customer', CheckKnownCustomer);

// 主线程调用
const result = await window.sharedStorage.worklet.run(
  'check-known-customer',
  { data: { userId: 'user-123' } }
);

if (result.isKnown) {
  console.log(`欢迎回来，${result.tier} 会员！`);
} else {
  console.log('欢迎新用户！');
}

class ABTestSelection extends SharedStorageSelectURLOperation {
  async run(data) {
    const { experimentId, forceVariant } = data;
    const key = `ab-test-${experimentId}`;
    
    // 如果强制指定变体
    if (forceVariant !== undefined) {

    }
    
    return parseInt(variant);
  }
}

// frequency-cap-worklet.js
class FrequencyCap extends SharedStorageRunOperation {
  async run(data) {
    const { adId, maxImpressions, timeWindow } = data;
    const key = `freq-${adId}`;
    
    // 获取当前展示记录
    const stored = await this.sharedStorage.get(key);
    let impressions = stored ? JSON.parse(stored) : [];
    
    // 过滤掉过期的记录
    const now = Date.now();
    const windowStart = now - timeWindow;
    impressions = impressions.filter(t => t > windowStart);
    
    // 检查是否超过限制
    if (impressions.length >= maxImpressions) {
      return { 
        allowed: false, 
        reason: 'frequency-cap-reached',
        currentCount: impressions.length 
      };
    }
    
    // 记录新的展示
    impressions.push(now);
    await this.sharedStorage.set(key, JSON.stringify(impressions));
    
    return { 
      allowed: true, 
      currentCount: impressions.length,
      maxImpressions 
    };
  }
}
register('frequency-cap', FrequencyCap);

// 主线程使用
async function showAdIfAllowed(adId) {
  const result = await window.sharedStorage.worklet.run(
    'frequency-cap',
    {
      data: {
        adId: adId,
        maxImpressions: 3,
        timeWindow: 24 * 60 * 60 * 1000 // 24小时
      }
    }
  );
  
  if (result.allowed) {
    console.log(`展示广告 (${result.currentCount}/${result.maxImpressions})`);
    // 展示广告逻辑...
  } else {
    console.log('广告已达到频次上限');
  }
}

// 设置用户偏好（在任意合作站点）
async function setUserPreference(preference, value) {
  await window.sharedStorage.set(
    `pref-${preference}`,
    JSON.stringify({
      value: value,
      updatedAt: Date.now()
    })
  );
}

// preference-worklet.js
class GetPreference extends SharedStorageRunOperation {
  async run(data) {
    const { preference, defaultValue } = data;
    const key = `pref-${preference}`;
    
    const stored = await this.sharedStorage.get(key);
    
    if (stored) {
      const parsed = JSON.parse(stored);
      return { 
        found: true, 
        value: parsed.value,
        updatedAt: parsed.updatedAt
      };
    }
    
    return { found: false, value: defaultValue };
  }
}
register('get-preference', GetPreference);

// 读取用户偏好（在任意合作站点）
async function getUserPreference(preference, defaultValue) {
  const result = await window.sharedStorage.worklet.run(
    'get-preference',
    { data: { preference, defaultValue } }
  );
  
  return result.value;
}

// aggregation-worklet.js
class ReportConversion extends SharedStorageRunOperation {
  async run(data) {
    const { campaignId, conversionType, value } = data;
    
    // 发送聚合报告
    await this.privateAggregation.sendHistogramReport({
      bucket: hashString(`${campaignId}-${conversionType}`),
      value: value
    });
    
    // 记录转化
    const key = `conversions-${campaignId}`;
    const stored = await this.sharedStorage.get(key);
    const conversions = stored ? parseInt(stored) : 0;
    await this.sharedStorage.set(key, String(conversions + 1));
    
    return { reported: true, totalConversions: conversions + 1 };
  }
}

// 辅助函数：将字符串转换为桶值
function hashString(str) {
  let hash = 0;
  for (let i = 0; i < str.length; i++) {
    const char = str.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;
  }
  return Math.abs(hash);
}

register('report-conversion', ReportConversion);

// 主线程使用
await window.sharedStorage.worklet.run(
  'report-conversion',
  {
    data: {
      campaignId: 'camp-001',
      conversionType: 'purchase',
      value: 100
    }
  }
);
