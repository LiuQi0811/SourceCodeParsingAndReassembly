// Worker 代码
self.onmessage = (event) => {
  const port = event.data.port;
  
  port.onmessage = (e) => {
    if (e.data.type === 'task') {
      const result = e.data.num * 2;
      port.postMessage({ result });
    } else if (e.data.type === 'heavy') {
      let sum = 0;
      for (let i = 0; i < e.data.iterations; i++) {
        sum += Math.sqrt(i);
      }
      port.postMessage({ result: sum });
    } else {
      port.postMessage('Worker收到: ' + e.data);
    }
  };
  port.postMessage('Worker已启动');
};
